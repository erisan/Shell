■数値判断
# 数値同士で演算を行った場合は「0」、数値同士でも計算結果が 0 になる場合は「1」、
# 数値以外で演算を行った場合(計算できない場合)は「2 以上」が終了ステータス
function IsNumeric() {
    if [ $# -ne 1 ]; then
         return 1
    fi
    expr "$1" + 1 >/dev/null 2>&1
    if [ $? -ge 2 ]; then
         return 1
    fi
    return 0
}


if IsNumeric $num
then
   export count=$num
else
   echo " $num is not Numeric"
   Usage
fi
